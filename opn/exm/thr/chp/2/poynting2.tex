\begin{question}[section=2,name={Poynting 2},difficulty=,quantity=5,type=thr,tags={20160310,20131024,20130625,20061016,20060816}]
	Wie sind die Poyntingvektoren $\vec{P}$ und $\vec{T}$ definiert? Wie berechnet man
 aus $\vec{T}$ die Wirkleistungsflussdichte / Blindleistungsflussdichte?
 \\ \textbf{Hinweis:}\\
 Skript Seite 15--16.
\end{question}
\begin{solution}
	\begin{align}
	\vec{P}(t) &= \vec{E}(t) \times \vec{H}(t)\\
\vec{T} &= \frac12 \vec{E} \times \vec{H}^* = \vec{T}_\mathrm{w} + j\vec{T}_\mathrm{b}
\end{align}
Die Wirkleistungsflussdichte $\vec{T}_\mathrm w = \Re\{\vec{T}\}$.\\
Die Blindleistungsflussdichte $\vec{T}_\mathrm b = \Im\{\vec{T}\}$.
\end{solution}
